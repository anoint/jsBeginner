# 새로안내용

## js로 style을 추가하는 법
  - dom객체.style.속성명 = 값;
ex - 공.style.fontSize = '25px';

## js로 class명 추가하는 법
ex - 공.className = '공아이디' + 숫자;
     공.classList.add(공아이디 + 숫자);
     
## js로 style에 값을 변수로 지정하는 법
ex - 공.style.background = 배경색;
